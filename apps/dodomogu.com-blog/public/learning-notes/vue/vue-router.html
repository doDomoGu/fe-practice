scrollBehavior ​

js
{
    routes: [
        //...
    ],
    scrollBehavior(to, from, savedPosition) {
        
        // 1. 滚动到绝对位置 x = ?, y = ?
        return { x:0, y:0 } // 滚动到开头
        // 2. 滚动到上一个保存的位置
        // 第三个参数savedPosition => 期望回到的位置
        if ( savedPosition ) {
         return savedPosition
        }
        // 3. 异步滚动
        return new Promise( resolve, reject ) => {
            setTimeout(() => {
                resolve({ x: 0, y: 0 })
            }, 1000)
        }
    }
}
