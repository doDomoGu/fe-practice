树 ​

定义 ​

  • 树是非线性数据结构
  • 每个节点都可能会有0个到多个后代
  • 每个节点(除了根节点)必须具备唯一父节点

  console.log(node.value)

    node.children.forEach(child => {

function dfs_BL(node) {
  const stack = [node]
  while(stack.length > 0) {
    const current = stack.pop()
    console.log(current.value)
    if(current.children){
      current.children.reverse().forEach(child=>{
        stack.push(child)
      })
    }
  }
}
